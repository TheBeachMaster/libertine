function install_crosscompiler_linux()
{
	pushdDependencyFolder
	
		make mrproper
		
		make V=2 O="$buildFolder" ARCH="$LFS_ARCHITECTURE" INSTALL_HDR_PATH=../../install/usr/include headers_install

		local -r buildFolder=../"$LFS_DEPENDENCY_NAME"-build
		mkdir -m 0755 -p "$buildFolder"
	
	popdDepdendencyFolder
}
